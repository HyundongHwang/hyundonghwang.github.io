# p69 버텍스 프래그먼트 쉐이더 작성
```
Shader "Unlit/SingleColor"
{
    Properties
    {
        _MainColor("MainColor", Color) = (1, 1, 1, 1)
    }
    SubShader
    {
        Pass
        {
            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            
            float4 vert (float4 vertex : POSITION) : SV_POSITION
            {
                return mul(UNITY_MATRIX_MAP, vertex);
            }
            
            fixed4 _MainColor;
            
            fixed4 frag () : SV_TARGET
            {
                return _MainColor;
            }
            ENDCG
        }
    }
}
```
